estate_natifs =
{
	icon = 1

	# If true, country will get estate
	trigger = {
		OR = {
			has_reform = illuminatos_council_reform
			tag = LGW
		}
	}

	# These scale with loyalty & power
	country_modifier_happy = {
		global_unrest = -1
	}
	country_modifier_neutral = {
		global_unrest = 1
	}	
	country_modifier_angry = {
		global_unrest = 2
	}

	land_ownership_modifier = {
		natifs_loyalty_modifier = 0.2
	}

	province_independence_weight = {
		factor = 1
		modifier = {
			factor = 0.1
			NOT = { culture_group = owner }
		}
		modifier = {
			factor = 0.5
			NOT = { culture = owner }
		}
		modifier = {
			factor = 0.67
			NOT = { is_state_core = owner }
		}
		modifier = {
			factor = 1.2
			base_tax = 10
		}
	}

	# Influence modifiers
	base_influence = 40
	influence_modifier = {
		desc = cultural_majority_natifs
		trigger = { 
			NOT = { dominant_culture = modern_french }
		}
		influence = 20
	}
	influence_modifier = {
		desc = cultural_minority_natifs
		trigger = { dominant_culture = modern_french }
		influence = -20
	}
	
	privileges = {
		estate_natifs_conscription
		estate_natifs_conscription_reduite
		estate_natifs_taxation_exceptionnelle
		estate_natifs_taxation_reduite
	}

	agendas = {
		estate_natifs_restore_currency
		estate_natifs_recovery_from_hyperinflation
		estate_natifs_get_cash
	}

	color = { 200 150 0 }

	influence_from_dev_modifier = 1.0
}
